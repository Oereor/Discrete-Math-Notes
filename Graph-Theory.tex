\chapter{Graph Theory}
\section{Basic Concepts}
\begin{definition}[Graph]
    A \emph{graph} is an ordered pair $(V, E)$, where $V$ is a set of vertices, and $E$ is a set of edges. The two vertices connected by a certain edge are called \emph{endpoints}. 
\end{definition}

Graphs are the abstraction of connectivity and intersection information. Note that the endpoints of an edge need not be distinct. In other words, an edge connecting some vertex to itself is allowed; such edge is called a \emph{loop}. 

We can sort graphs into different categories based on the properties of their vertices and edges. For example, by whether the edges are directed or not, by whether there are multiple edges between the same pair of vertices, or by having loops or not. Specifically, we have the following definition of a simple graph. 

\begin{definition}[Simple Graph]
    A graph is said to be \emph{simple}, if it is undirected and does not have multiple edges or loops.
\end{definition}

Sometimes we would use the term \emph{simple directed graph}, which refers to the graph that meets the requirements of a simple graph, except being undirected. 

We now introduce the concept of degree of a vertex.

\begin{definition}
    Given an \emph{undirected} graph $G=(V,E)$, two vertices $u,v\in V$ are called \emph{adjacent} (or \emph{neighbors}) if there exists an edge $e\in E$ connecting them. Such $e$ is called \emph{incident} with $u,v$. 

    The \emph{neighborhood} of a vertex $v$, denoted $\mathcal{N}(v)$, is the set of all neighbors of $v$. If $A\subseteq V$, then $\mathcal{N}(A)$ is defined as $\displaystyle\mathcal{N}(A)=\bigcup_{v\in A}\mathcal{N}(v)$. 
\end{definition}

\begin{definition}[Degree]
    Given an undirected graph $G=(V,E)$, the \emph{degree} of some $v\in V$ is the number of edges incident with $v$. Any loop contributes twice to the degree of its vertex. The degree of $v$ is denoted by $\deg(v)$.
\end{definition}

Note that we have an alternative method to calculate the degree of $v$: 
\[\deg(v)=|\{e\mid v\text{ is the first endpoint}\}|+|\{e\mid v\text{ is the second endpoint}\}|.\]
Thus we immediately have the following theorem, also known as \emph{Handshaking Theorem}. 

\begin{theorem}
    If $G=(V,E)$ is undirected, then $\displaystyle\sum_{v\in V}\deg(v)=2|E|$. 
\end{theorem}

In other words, in an undirected graph, the sum of the degrees of all vertices is twice the number of edges. 

\begin{proof}
    Since we can write 
    \[\deg(v)=\sum_{e\in E}\left( \mathbf{1}_{\{v\text{ is the first endpoint of }e\}} + \mathbf{1}_{\{v\text{ is the second endpoint of }e\}} \right),\] 
    we have 
    \begin{align*}
        \sum_{v\in V}\deg(v)&=\sum_{v\in V}\sum_{e\in E}\left( \mathbf{1}_{\{v\text{ is the first endpoint of }e\}} + \mathbf{1}_{\{v\text{ is the second endpoint of }e\}} \right)\\
        &=\sum_{e\in E}\sum_{v\in V}\left( \mathbf{1}_{\{v\text{ is the first endpoint of }e\}} + \mathbf{1}_{\{v\text{ is the second endpoint of }e\}} \right)\\
        &=2|E|.
    \end{align*}
\end{proof}

\begin{corollary}
    In an undirected graph, the number of vertices with odd degree is even.
\end{corollary}

We point out some key terminologies in directed graphs that are different from those in undirected graphs. 

\begin{definition}[Directed Edge]
    If $G=(V, E)$ is a directed graph and $e\in E$ is from $u$ to $v$, then:
    \begin{itemize}
        \item $u$ is adjacent \emph{to} $v$, and $v$ is adjacent \emph{from} $u$;
        \item $u$ is the \emph{initial} vertex of $e$, and $v$ is the \emph{terminal} (or \emph{end}) vertex of $e$.
    \end{itemize}
\end{definition}

We also have the concepts of \emph{in-degree} and \emph{out-degree} for vertices in directed graphs.

\begin{definition}[In- and Out-Degrees]
    If $G=(V, E)$ is a directed graph and $v\in V$, we define
    \begin{itemize}
        \item (In-degree) $\deg^-(v):=|\{e\in E\mid e\text{ is associated with }(u',v')\text{ and }v=v'\}|$;
        \item (Out-degree) $\deg^+(v):=|\{e\in E\mid e\text{ is associated with }(u',v')\text{ and }v=u'\}|$.
    \end{itemize}
\end{definition}

The relationship between in-degrees and out-degrees is given by the following theorem.

\begin{theorem}
    If $G=(V,E)$ is a finite directed graph, then \[\sum_{v\in V}\deg^-(v)=\sum_{v\in V}\deg^+(v)=|E|.\]
\end{theorem}

\section{Paths and Circuits}
\begin{definition}[Path]
    Suppose $G=(V,E)$ is an undirected graph. A \emph{path} of length $n\ (n\geq 0)$ from $u$ to $v$ is a finite sequence of edges $e_1,e_2,\dots,e_n$ such that there exists a sequence of vertices $v_0,v_1,\dots,v_n$ with $v_0=u$, $v_n=v$, and for each $i=1,2,\dots,n$, $v_{i-1}$ and $v_i$ are the endpoints of $e_i$.

    A path is called \emph{simple} if it does not contain duplicate edges. We say that the path $e_1,\dots,e_n$ \emph{traverses} these edges, and \emph{passes through} the vertices $v_0,v_1,\dots,v_n$.
\end{definition}

For simplicity, when the graph is simple, we may also write the path as a sequence of the vertices $v_0,v_1,\dots,v_n$ that it passes through. 

\begin{definition}[Circuit]
    A \emph{circuit} is a path of positive length, whose endpoints are the same vertex. A circuit is also called \emph{simple} if it does not contain duplicate edges.
\end{definition}

Note that a simple path does not necessarily pass through distinct vertices. Recall the definition of a loop, and we see that \emph{a loop is a path of length $1$; moreover, it is a circuit.}

We can define path and circuits in directed graphs in a similar way, with the additional requirement that for each $i=1,2,\dots,n$, $e_i$ is a directed edge from $v_{i-1}$ to $v_i$.

\section{Connectivity}\label{sec:Connectivity}
In this section we shall talk about connectivity of graphs. The behavior of connectedness is quite different between directed and undirected graphs, and we will begin with the simpler one, i.e., undirected graphs. 

\begin{definition}
    Suppose $G=(V,E)$ is an undirected graph. Two vertices $u,v\in V$ are \emph{connected} if there is a path from $u$ to $v$ in $G$. 
\end{definition}

Note that in this definition, we did not explicitly require the path to be simple. In fact, if there is a path from $u$ to $v$, then there is also a simple path from $u$ to $v$. We shall not prove this fact here, but in future discussions we will make use of it. 

The following theorem indicates that connectedness is an equivalence relation.

\begin{theorem}
    In an undirected graph, connectedness is an equivalence relation. That is, 
    \begin{enumerate}[itemindent=0ex, itemsep=0pt, label=\upshape{(\alph*)}]
        \item (Reflexive) Any vertex is connected to itself;
        \item (Symmetric) If $u$ is connected to $v$, then $v$ is connected to $u$;
        \item (Transitive) If $u$ is connected to $v$ and $v$ is connected to $w$, then $u$ is connected to $w$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    For any vertex $v$, there is a path of length $0$ from $v$ to itself, so (a) holds.

    If there is a path from $u$ to $v$, then we can reverse the path to obtain a path from $v$ to $u$. In symbols, if $e_1,\dots,e_n$ is a path from $u$ to $v$, then $e_n,\dots,e_1$ is a path from $v$ to $u$. Hence (b) holds.

    If there is a path from $u$ to $v$ and a path from $v$ to $w$, we can concatenate these two paths to obtain a path from $u$ to $w$. In symbols: If $e_1,\dots,e_n$ is a path from $u$ to $v$, and $f_1,\dots,f_m$ is a path from $v$ to $w$, then $e_1,\dots,e_n,f_1,\dots,f_m$ is a path from $u$ to $w$. This proves (c).
\end{proof}

With the connectedness of vertices defined, we can define the connectedness of the whole graph.

\begin{definition}[Connected Graph]
    An undirected graph $G=(V,E)$ is \emph{connected} if every pair of vertices in $V$ are connected. Otherwise, it is \emph{disconnected}.
\end{definition}

To further explore the properties of connected graphs, we shall define some concepts first. 

\begin{definition}[Subgraph]
     A \emph{subgraph} of a graph $G=(V,E)$ is a graph $G'=(V',E')$ such that $V'\subseteq V$ and $E'\subseteq E$. $G'$ is said to be \emph{proper} if $G'\neq G$. 
\end{definition}

\begin{definition}[Induced Subgraph]
    Suppose $G=(V,E)$ is a graph and $W\subseteq V$ is a subset of vertices. The subgraph \emph{induced} by $W$ consists of all vertices in $W$ and all edges in $E$ whose endpoints both lie in $W$. 
\end{definition}

\begin{definition}[Connected Component]
    A \emph{connected component} of a graph $G=(V,E)$ is a maximal connected subgraph of $G$. Here ``maximal'' means that the subgraph is not a proper subgraph of any other connected subgraph of $G$.
\end{definition}

We now discuss an important theorem regarding connected components.

\begin{theorem}\label{thm:theorem about connected components}
    Suppose $G$ is a nonempty undirected graph, and $G'$ is a subgraph of $G$. Then $G'$ is a connected component of $G$ if and only if there is a vertex $v$ such that $G'$ is the subgraph induced by the set of vertices connected to $v$, namely, the set $W=\{u\mid u\text{ is connected to }v\}$. 
\end{theorem}

\begin{proof}
    Assume $G'=(V', E')$ is a connected component of $G$. We can pick any vertex $v$ in $G'$ and construct $W$ with this $v$. Now we first show that $V'=W$. 

    If $v'\in V'$ and $v'\not\in W$, then $v'$ is not connected to $v$, which contradicts the fact that $G'$ is connected. So $V'\subseteq W$. If $u\in W$ and $u\not\in V'$, then $u$ is connected to $v$, so there is a path from $u$ to $v$. By adding this path to $G'$, we can obtain a connected subgraph that contains $G'$ as a proper subgraph, which contradicts the maximality of $G'$. So $W\subseteq V'$. Hence $V'=W$.

    Then we prove that $G'$ is the subgraph induced by $W$. If there is an edge $e$ in $E$ whose endpoints both lie in $W$ but $e\not\in E'$, then we can add $e$ to $G'$ to obtain a connected subgraph that contains $G'$ as a proper subgraph. This yields a contradiction. 

    Conversely, suppose $G'$ is the subgraph induced by $W$. Then $G'$ is connected since any two vertices in $W$ are connected (by the transitive property) in $G$. Since $G'$ contains all edges in $G$ that connects the vertices in $W$, the vertices are also connected in $G'$. In symbols, if $u_1$ and $u_2$ are both in $G'$, then there is a path $e_1,\cdots,e_n$ from $u_1$ to $u_2$ in $G$, and since all vertices in this path are in $W$, all edges in this path are in $G'$. Hence there is a path from $u_1$ to $u_2$ in $G'$.

    Moreover, if there is a connected subgraph $H$ that contains $G'$ as a proper subgraph, then there is a vertex in $H$ (so that it is connected to $v$) that is not in $G'$, which contradicts the definition of $W$. Hence $G'$ is a connected component of $G$.
\end{proof}

Now we turn to the connectedness of directed graphs. The situation is more complicated here, but we can still define a type of connectedness that is similar to the one in undirected graphs.

\begin{definition}[Strongly Connected]
    Suppose $G=(V,E)$ is a directed graph. Two vertices $u,v\in V$ are \emph{strongly connected} if there is a path from $u$ to $v$ and a path from $v$ to $u$ in $G$.
\end{definition}

In directed graphs, instead of saying connectedness of two vertices, we say that $v\in V$ is \emph{reachable} from $u\in V$ if there is a path from $u$ to $v$; note that here the order of vertices matters. Strongly connected vertices are also called \emph{mutually reachable} vertices. We can also define the concept of \emph{strongly connected graphs} and \emph{strongly connected components} in a similar way as we did for undirected graphs, which we shall not do here to avoid redundancy.

It is easy to see that strongly connectedness is also an equivalence relation. The proof is quite similar to its counterpart in undirected graphs, hence we shall not present the details here. 

\cref{thm:theorem about connected components} also holds for strongly connected components in directed graphs, with slight modifications in statement. We now present the directed version of the theorem without proof.

\begin{theorem}
    Suppose $G$ is a nonempty directed graph, and $G'$ is a subgraph of $G$. Then $G'$ is a strongly connected component of $G$ if and only if there is a vertex $v$ such that $G'$ is the subgraph induced by the set of vertices strongly connected to $v$, namely, the set $W=\{u\mid u\text{ is strongly connected to }v\}$.
\end{theorem}

Basically what we did was to replace the term ``connected'' with ``strongly connected'' in \cref{thm:theorem about connected components}.

The following theorem depicts the connectedness between strongly connected components. 

\begin{theorem}
    Suppose $G$ is a directed graph, $G'=(V',E')$ and $G''=(V'',E'')$ are two distinct strongly connected components of $G$. Then for any $u_1,u_2\in V'$ and any $v_1,v_2\in V''$, $v_1$ is reachable from $u_1$ if and only if $v_2$ is reachable from $u_2$.
\end{theorem}

This theorem indicates that if there is a path from some vertex in $G'$ to some vertex in $G''$, then there is a path from any vertex in $G'$ to any vertex in $G''$. 

\begin{proof}
    If $v_1$ is reachable from $u_1$, then there is a path from $u_1$ to $v_1$. Since $u_2$ is strongly connected to $u_1$ and $v_2$ is strongly connected to $v_1$, there is a path from $u_2$ to $u_1$ and a path from $v_1$ to $v_2$. By concatenating these paths, we can obtain a path from $u_2$ to $v_2$. 
     
    Proof for the converse is similar. 
\end{proof}

\begin{theorem}
    If $G$ is a directed graph and $e_1,\dots,e_n$ is a circuit passing through $x_0,x_1,\dots,x_n=x_0$, then $x_0,x_1,\dots,x_n$ belong to the same strongly connected component of $G$.
\end{theorem}

In other words, there is no circuit that passes through two distinct strongly connected components. 

\begin{proof}
    We prove by contradiction. Assume $x_0,x_1,\dots,x_n$ do not belong to the same strongly connected component. Then there exist two vertices in this circuit that are not strongly connected. However, since the path $x_0,\dots,x_n$ is a circuit, there must exist a path between any two vertices $(x_i,x_j)$, where $x_i$ is the initial vertex of the path and $x_j$ is the terminal vertex. By exchanging the roles of $x_i$ and $x_j$, we obtain a path from $x_j$ to $x_i$, which contradicts the assumption that $x_i$ and $x_j$ are not strongly connected. 
\end{proof}

Now we consider the effects of adding or removing edges in a graph, as described by the following theorem. 

\begin{theorem}
    Suppose $G=(V,E)$ and $G'=(V,E\cup\{e_0\})$, where $e_0\not\in E$ and $e_0$ connects the vertices $u,v$. Then
    \begin{enumerate}[itemindent=0ex, itemsep=0pt, label=\upshape{(\alph*)}]
        \item for any $x,y\in V$, if $x$ and $y$ are connected in $G$, then they are also connected in $G'$. 
        \item if $u$ and $v$ are connected, then for any $x,y\in V$, $x$ and $y$ are connected in $G$ if and only if they are connected in $G'$. 
        \item if $u$ and $v$ are not connected in $G$, then for any $x\in V$, $x$ is connected to $u$ in $G'$ if and only if $x$ is connected to $v$ in $G'$; if and only if $x$ is connected to $u$ or $v$ in $G$. 
        \item for any $x,y\in V$, if $x$ and $y$ are not connected in $G$ but connected in $G'$, then $x,y,u,v$ are connected in $G'$. 
    \end{enumerate}
\end{theorem}

\begin{proof}
    Part (a) is trivial, since $G$ is a subgraph of $G'$. 

    For part (b), one direction follows directed from part (a). For the converse, suppose $x$ and $y$ are connected in $G'$. Then there is a path from $x$ to $y$ in $G'$. If this path does not use the edge $e_0$, then it is also a path in $G$, so $x$ and $y$ are connected in $G$. Otherwise the path uses $e_0$, say it goes from $x$ to $u$ to $v$ to $y$ (or $x$ to $v$ to $u$ to $y$, which is basically the same), then since $u$ and $v$ are connected in $G$, we can replace the edge $e_0$ with a path from $u$ to $v$ in $G$ to obtain a path from $x$ to $y$ in $G$. Hence $x$ and $y$ are connected in $G$.

    Proof of part (c) includes 4 different directions, 3 of which are trivial. The only hard direction is to prove that if $x$ is connected to $u$ in $G'$, then $x$ is connected to $u$ or $v$ in $G$. 

    By the assumption, there exists a path from $x$ to $u$ in $G'$. If the path does not contain $e_0$ then we are done. Otherwise, suppose the path is $e_1,\dots,e_n$, passing through $x_0=x, x_1, \dots, x_m=u$, and $e_j=e_0$ is the first time that $e_0$ appears in the sequence. Then either $x_{j-1}=u$ or $x_{j-1}=v$, and since $x_{j-1}$ is connected to $x$, we see that $x$ is connected either to $u$ or to $v$. 

    Part (d) follows from the fact that the path from $x$ to $y$ in $G'$ must contain $e_0$. 
\end{proof}

\section{Trees}
Trees are a special type of graphs. 

\begin{definition}[Tree]
    A \emph{tree} is a connected undirected graph that does not contain any simple circuit.
\end{definition}

In \cref{sec:Connectivity} we stated the following fact without proof. Now we prove it here. 

\begin{lemma}
    If there exists a path from $u$ to $v$, then there must also exist a simple path from $u$ to $v$. 
\end{lemma}

\begin{proof}
    Since the length of paths are bounded below, among all paths from $u$ to $v$, there must exist a shortest one. If this one is not simple, say it's $e_1,\dots,e_n$ with $e_i=e_j=e_0\ (i\neq j)$. Then we can just erase $e_0$ from the path. Here ``erasing'' can be divided into two cases: $x_{i-1}=x_{j-1}$, or $x_{i-1}=x_j$. In the first case, we simply connect $x_{i-1}$ with $x_j$; in the second case, we connect $x_{i-1}$ with $x_{j+1}$. Hence in either case, we can obtain a shorter path from $u$ to $v$, contradicting our assumption. 
\end{proof}

The following theorem depicts a key property of trees, i.e., it is the ``maximal'' connected graph without circuits, and ``minimal'' connected graph.

\begin{theorem}\label{thm:key property of trees}
    If we add an edge to a tree, the graph will have a simple circuit; if we remove an edge from a tree, the graph will become disconnected. 
\end{theorem}

\begin{proof}
    If we add an edge between $u,v$, then since $u$ and $v$ are already connected, we can get a path $u,\dots,v,u$, which is a circuit. 

    If we remove an edge between $u,v$ and keep the graph connected, then by the first part of our proof, adding the edge back will create a simple circuit in the tree, which is a contradiction. 
\end{proof}

Trees can be categorized into \emph{rooted tree} and \emph{unrooted tree}, by whether a vertex is designated to be the root. We shall discuss unrooted tree first. 

\begin{definition}[Leaf]
    A vertex is a \emph{leaf} in a tree if its degree is $1$. 
\end{definition}

The next theorem guarentees the existence of leaf in trees. 

\begin{theorem}\label{thm:tree has at least one leaf}
    Every tree with at least two vertices has at least one leaf.
\end{theorem}

\begin{proof}
    Since trees are connected, we can pick any vertex $v$ and find a longest path from $v$ to some vertex $u$. Then $u$ is a leaf, since if it is not, we can extend the path to get a longer one, which contradicts the assumption.
\end{proof}

\begin{corollary}
    Every tree with at least two vertices has at least two leaves.
\end{corollary}

\begin{proof}
    In the proof of \cref{thm:tree has at least one leaf} we found a leaf $u$. Now we do the same operation again with $u$ in the place of $v$ to find another leaf. Clearly this new leaf cannot be $u$ since no circuit exists in a tree. Hence we have found two distinct leaves.
\end{proof}

We have the following theorem that characterizes trees in terms of the number of vertices and edges.

\begin{theorem}
    If $G=(V,E)$ is a tree, then $|E|=|V|-1$.
\end{theorem}

\begin{proof}
    We prove by induction on the number of vertices. 
    
    If $|V|=1$, then $|E|=0$, so the statement holds. 
    
    Assume the statement holds for any tree with less than $n$ vertices, where $n\geq 2$. Now we consider a tree $G$ with $n$ vertices. By \cref{thm:tree has at least one leaf}, there is a leaf $v$. Let $u$ be the unique neighbor of $v$. Then we can remove the edge between $u$ and $v$, and remove the vertex $v$ to obtain a new graph $G'=(V',E')$. It is easy to see that $G'$ is still a tree, and $|V'|=n-1$. By the induction hypothesis, we have $|E'|=|V'|-1=n-2$. Since we removed one edge to get from $G$ to $G'$, we have $|E|=|E'|+1=n-1=|V|-1$.
\end{proof}

In fact, the converse of \cref{thm:key property of trees} also holds, which gives us another characterization of trees.

\begin{theorem}
    An undirected graph with no simple circuit is a tree if adding an arbitrary edge generates a simple circuit in the new graph; a connected undirected graph is a tree if removing any edge makes the new graph disconnected.
\end{theorem}

\begin{proof}
    Assume $G=(V,E)$ is an undirected graph with no simple circuit, and for any $u,v\in V$, if we add an edge between $u$ and $v$, denoted $e_0$, then there is a simple circuit in the new graph. We wish to show that $u$ and $v$ are connected. Since $G$ contains no simple circuit, the simple circuit in the new graph must contain $e_0$. Removing $e_0$ from this circuit, we get a path from $u$ to $v$ in $G$. Hence $G$ is connected, and thus a tree.

    Next we assume $G=(V,E)$ is a connected undirected graph, and if we remove any edge in $G$, the new graph becomes disconnected. To prove that $G$ is a tree, it suffices to show that $G$ does not contain any simple circuit. If not so, assume $e_1,\dots,e_n$ is a simple circuit in $G$, passing through $v_0,v_1,\dots,v_n=v_0$. Then we can remove the edge $e_1$ to obtain a new graph. Since there is still a path from $v_0$ to $v_1$ in the new graph, it is still connected, which is a contradiction. 
\end{proof}

To summarize, the following statements are equivalent for an undirected graph $G=(V,E)$:
\begin{enumerate}[itemindent=0ex, itemsep=0pt, label=\upshape{(\alph*)}]
    \item $G$ is a tree, i.e. $G$ is connected but contains no simple circuits.
    \item $G$ is connected, and $|E|=|V|-1$.
    \item $G$ contains no simple circuits and $|E| = |V|-1$.
    \item $G$ is connected, but every edge in $E$ is a cut edge.
    \item $G$ contains no simple circuits, but adding any edge to $G$ generates a simple circuit.
    \item There is a unique simple path between any two vertices in $G$.
\end{enumerate}

We now discuss a special type of trees known as spanning trees. 

\begin{definition}[Spanning Tree]
    A \emph{spanning tree} of an undirected graph $G=(V,E)$ is a subgraph of $G$ that is a tree and contains all vertices in $V$.
\end{definition}

\begin{theorem}
    If $G$ is an undirected graph, then $G$ has a spanning tree if and only if $G$ is connected.
\end{theorem}

\begin{proof}
    If $G$ has a spanning tree, then $G$ is connected since the spanning tree is connected. 

    Conversely, if $G$ is connected, we can construct a spanning tree by starting with an empty graph and adding edges one by one. We can add an edge if and only if it does not create a simple circuit in the new graph. Since $G$ is connected, we can keep adding edges until all vertices are connected. The resulting graph is a spanning tree of $G$.
\end{proof}

In applicative scenarios, we often care more about the spanning tree with some optimality, such as the ``shortest'' one. This leads to the concept of minimum spanning tree, which is defined in a weighted graph. 

\begin{definition}[Weighted Undirected Graph]
    A \emph{weighted undirected graph} is an undirected graph $G=(V,E)$ together with a weight function $w\colon E\to\mathbb{R}$ that assigns to each edge a real number.
\end{definition}

\begin{definition}[Minimum Spanning Tree]
    Suppose $G=(V,E)$ is a connected weighted graph. A \emph{minimum spanning tree} of $G$ is a spanning tree $T$ of $G$ such that the total weight of edges in $T$ is the minimum among all spanning trees of $G$. In symbols, if $T'=(V,E')$ is any spanning tree of $G$, then \[\sum_{e\in E}w(e)\leq \sum_{e\in E'}w(e).\]
\end{definition}

There are two well-known algorithms for finding a minimum spanning tree, Prim's and Kruskal's. 

\begin{theorem}[Prim's Algorithm]
    A minimum spanning tree of $G=(V,E)$ can be constructed by following the steps below:
    \begin{enumerate}[itemindent=0ex, itemsep=0pt, label=\upshape{(\alph*)}]
        \item Choose the initial tree $T_0=(V_0,E_0)$ as an arbitrary singleton vertex with no edge.
        \item Construct $T_{n+1}$ from $T_n$ by adding an edge $e$ with the minimum weight among all remaining edges $e'$ satisfying that $T_n$ added with $e'$ is still a tree.
        \item Finally the graph $T_{|V|-1}$ is a minimum spanning tree of $G$.
    \end{enumerate}
\end{theorem}

\begin{proof}
    We first show that $T_{|V|-1}$ is a spanning tree of $G$. Since we add edges one by one, $T_{|V|-1}$ is a tree. Moreover, since we keep adding edges until all vertices are connected, $T_{|V|-1}$ contains all vertices in $V$. Hence $T_{|V|-1}$ is a spanning tree of $G$.

    We first prove a proposition: For any $0\leq n\leq |V|-1$, there exist a minimum spanning tree $T^*$ of $G$ such that $T_n$ is a subgraph of $T^*$. We prove this proposition by induction on $n$.

    If $n=0$, then $T_0$ is a singleton vertex, which is a subgraph of any spanning tree of $G$.

    Assume the proposition holds for $n\geq 0$. Now we consider $n+1$. By the induction hypothesis, there is a $T^*$ such that $T_n$ is a subgraph of $T^*$. If $e$ is an edge in $T^*$, then we are done. 
    Otherwise, since $T^*$ is a spanning tree, there is a path in $T^*$ connecting the endpoints of $e$. This path must contain an edge $e'$ that is not in $T_n$, since otherwise adding $e$ to $T_n$ will create a simple circuit, which contradicts the definition of $e$. Now we can replace $e'$ with $e$ to obtain a new spanning tree. Since Prim's algorithm picked $e$, we know that $w(e)\leq w(e')$, hence this new spanning tree is also a minimum spanning tree. Moreover, it contains $T_{n+1}$ as a subgraph. Hence the proposition holds for $n+1$.

    Finally, by the proposition for $n=|V|-1$, there is a minimum spanning tree that contains $T_{|V|-1}$ as a subgraph. Since both of them are spanning trees, they must be the same graph. Hence $T_{|V|-1}$ is a minimum spanning tree of $G$.
\end{proof}

\begin{theorem}[Kruskal's Algorithm]
    A minimum spanning tree of $G=(V,E)$ can be constructed by following the steps below:
    \begin{enumerate}[itemindent=0ex, itemsep=0pt, label=\upshape{(\alph*)}]
        \item Sort the edges in $E$ in non-decreasing order of their weights, and denote the sorted edges by $e_1,e_2,\dots,e_{m}$, where $m=|E|$. Initialize $T=\emptyset$.
        \item For each $i=1,2,\dots,m$, add $e_i$ to $T$ if doing so does not create a simple circuit in $T$.
        \item Stop when there are $|V|-1$ edges in $T$. The graph $T$ is then a minimum spanning tree of $G$.
    \end{enumerate}
\end{theorem}

Proof of Kruskal's algorithm is very similar to that of Prim's, so we will not enter a detailed discussion here.
